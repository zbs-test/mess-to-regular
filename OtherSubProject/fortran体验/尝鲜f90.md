# 体验
// main.f90
```fortran
PROGRAM add_numbers
REAL :: a, b, av1, av2
PRINT *,'input two numbers: '
READ (*,*) a, b
	av1 = (a+b)/2;av2=(a*b)**0.5
	PRINT *,'Arithmetic mean: ',av1
	PRINT *,'Geometric mean: ',av2
END
```
# 编译
//cmd
`gfortran main.f90`
